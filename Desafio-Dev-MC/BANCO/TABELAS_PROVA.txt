SET SQL DIALECT 3;

CREATE TABLE CLIENTE (
    CODIGO              CODIGO NOT NULL /* CODIGO = INTEGER */,
    NOME                NOME /* NOME = VARCHAR(35) */,
    ENDERECO            ENDERECO /* ENDERECO = VARCHAR(35) */,
    BAIRRO              BAIRRO /* BAIRRO = VARCHAR(20) */,
    CIDADE              CIDADE /* CIDADE = VARCHAR(30) */,
    CEP                 CEP /* CEP = VARCHAR(10) */,
    UF                  UF /* UF = CHAR(2) */,
    FONE                TELEFONE /* TELEFONE = VARCHAR(15) */,
    CELULAR             TELEFONE /* TELEFONE = VARCHAR(15) */,
    EMAIL               VARCHAR(50)
);
ALTER TABLE CLIENTE ADD CONSTRAINT PK_CLIENTE PRIMARY KEY (CODIGO);



CREATE TABLE PRODUTOS (
    CODIGO                    CODIGO NOT NULL /* CODIGO = INTEGER */,
    DESCRICAO                 VARCHAR(35),
    REFERENCIA                VARCHAR(15),
    UNIDADE                   VARCHAR(4),
    DATA_VENDA                EMISSAO /* EMISSAO = DATE */,
    PRECO_VENDA               DINHEIRO /* DINHEIRO = NUMERIC(15,4) DEFAULT 0 */,
    SALDO                     DINHEIRO /* DINHEIRO = NUMERIC(15,4) DEFAULT 0 */    
)    ;
ALTER TABLE PRODUTOS ADD CONSTRAINT PK_PRODUTOS PRIMARY KEY (CODIGO);



CREATE TABLE VENDAS (
    NRNOTA                CODIGO NOT NULL /* CODIGO = INTEGER */,
    EMISSAO               EMISSAO /* EMISSAO = DATE */,
    CLIENTE               CODIGO NOT NULL /* CODIGO = INTEGER */,
    OPERACAO_VENDA        TIPODIVERSO /* TIPODIVERSO = CHAR(1) */,
    TIPO_VENDA            TIPODIVERSO /* TIPODIVERSO = CHAR(1) */,
    VALOR_VENDA           DINHEIRO /* DINHEIRO = NUMERIC(15,4) DEFAULT 0 */
)    ;
ALTER TABLE VENDAS ADD CONSTRAINT PK_VENDAS PRIMARY KEY (NRNOTA);
    


CREATE TABLE ITEM_VENDA (
    LCTO             INTEGER NOT NULL,
    NR_VENDA         CODIGO /* CODIGO = INTEGER */,
    PRODUTO          CODIGO /* CODIGO = INTEGER */,
    QTDE             QUANTIDADE /* QUANTIDADE = NUMERIC(15,4) DEFAULT 0 */,
    VALOR_UNIT       DINHEIRO /* DINHEIRO = NUMERIC(15,4) DEFAULT 0 */,
    VALOR_TOTAL      DINHEIRO /* DINHEIRO = NUMERIC(15,4) DEFAULT 0 */
    
);
ALTER TABLE ITEM_VENDA ADD CONSTRAINT PK_ITEM_VENDA PRIMARY KEY (LCTO);
